import { useState, useEffect, useCallback } from 'react';
import { prendasAPI } from './services/api';
import PrendaCard from './components/PrendaCard';
import ModalEditarPrenda from './components/ModalEditarPrenda';
import ModalLogin from './components/ModalLogin';
import AdminPanel from './components/AdminPanel';

function App() {
  const [prendas, setPrendas] = useState([]);
  const [prendaSeleccionada, setPrendaSeleccionada] = useState(null);
  const [mostrarModalEditar, setMostrarModalEditar] = useState(false);
  const [mostrarModalLogin, setMostrarModalLogin] = useState(false);
  const [usuario, setUsuario] = useState(null);
  const [cargando, setCargando] = useState(true);
  const [mensaje, setMensaje] = useState({ texto: '', tipo: '' });
  const [carrito, setCarrito] = useState([]);
  const [busqueda, setBusqueda] = useState('');
  const [categoriaFiltro, setCategoriaFiltro] = useState('');
  const [mostrarAdminPanel, setMostrarAdminPanel] = useState(false);

  const mostrarMensaje = useCallback((texto, tipo = 'success') => {
    setMensaje({ texto, tipo });
    setTimeout(() => setMensaje({ texto: '', tipo: '' }), 3000);
  }, []);

  const cargarPrendas = useCallback(async () => {
    try {
      const respuesta = await prendasAPI.obtenerTodas();
      setPrendas(respuesta.prendas);
    } catch {
      mostrarMensaje('Error al cargar las prendas', 'error');
    } finally {
      setCargando(false);
    }
  }, [mostrarMensaje]);

  useEffect(() => {
    cargarPrendas();
    const usuarioGuardado = localStorage.getItem('usuario');
    if (usuarioGuardado) {
      setUsuario(JSON.parse(usuarioGuardado));
    }
    // Cargar carrito del localStorage
    const carritoGuardado = localStorage.getItem('carrito');
    if (carritoGuardado) {
      setCarrito(JSON.parse(carritoGuardado));
    }
  }, [cargarPrendas]);

  // Guardar carrito en localStorage cuando cambie
  useEffect(() => {
    localStorage.setItem('carrito', JSON.stringify(carrito));
  }, [carrito]);

  const handleEditar = (prenda) => {
    setPrendaSeleccionada(prenda);
    setMostrarModalEditar(true);
  };

  const handleGuardarEdicion = async (id, datos) => {
    try {
      const respuesta = await prendasAPI.actualizar(id, datos);
      setPrendas(prendas.map(p => p.id === id ? respuesta.prenda : p));
      setMostrarModalEditar(false);
      setPrendaSeleccionada(null);
      mostrarMensaje('Prenda actualizada exitosamente', 'success');
    } catch (error) {
      mostrarMensaje(error.message || 'Error al actualizar la prenda', 'error');
    }
  };

  const handleComprar = async (prenda) => {
    if (!usuario) {
      setMostrarModalLogin(true);
      return;
    }

    // Agregar al carrito
    const itemExistente = carrito.find(item => item.id === prenda.id);
    if (itemExistente) {
      setCarrito(carrito.map(item =>
        item.id === prenda.id
          ? { ...item, cantidad: item.cantidad + 1 }
          : item
      ));
    } else {
      setCarrito([...carrito, { ...prenda, cantidad: 1 }]);
    }
    mostrarMensaje('Producto agregado al carrito', 'success');
  };

  const handleLoginExitoso = (usuarioData) => {
    setUsuario(usuarioData);
    mostrarMensaje('SesiÃ³n iniciada correctamente', 'success');
  };

  const handleCerrarSesion = () => {
    localStorage.removeItem('token');
    localStorage.removeItem('usuario');
    setUsuario(null);
    mostrarMensaje('SesiÃ³n cerrada', 'success');
  };

  // Filtrar prendas por bÃºsqueda y categorÃ­a
  const prendasFiltradas = prendas.filter(prenda => {
    const coincideBusqueda = prenda.nombre.toLowerCase().includes(busqueda.toLowerCase()) ||
                             prenda.descripcion.toLowerCase().includes(busqueda.toLowerCase());
    const coincideCategoria = !categoriaFiltro || prenda.categoria === categoriaFiltro;
    return coincideBusqueda && coincideCategoria;
  });

  // Obtener categorÃ­as Ãºnicas
  const categorias = [...new Set(prendas.map(p => p.categoria))];

  if (cargando) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-gray-900">
        <div className="text-xl text-white">Cargando prendas...</div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-900 text-white">
      {/* Header */}
      <header className="bg-gray-900 border-b border-gray-800">
        <div className="container mx-auto px-4 py-4 flex justify-between items-center">
          <h1 className="text-3xl font-bold bg-linear-to-r from-pink-500 to-purple-500 bg-clip-text text-transparent">
            Mera's
          </h1>
          <nav className="hidden md:flex items-center gap-6">
            <a href="#catalogo" className="text-white hover:text-pink-400 transition-colors">CatÃ¡logo</a>
            <a href="#categorias" className="text-white hover:text-pink-400 transition-colors">CategorÃ­as</a>
            <a href="#contacto" className="text-white hover:text-pink-400 transition-colors">Contacto</a>
          </nav>
          <div className="flex items-center gap-4">
            <button
              onClick={() => setMostrarAdminPanel(true)}
              className="bg-gray-800 hover:bg-gray-700 text-white font-semibold py-2 px-4 rounded-lg transition-colors duration-200"
            >
              Admin
            </button>
            <div className="relative">
              <button className="text-white hover:text-pink-400 transition-colors">
                <svg className="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M3 3h2l.4 2M7 13h10l4-8H5.4M7 13L5.4 5M7 13l-2.293 2.293c-.63.63-.184 1.707.707 1.707H17m0 0a2 2 0 100 4 2 2 0 000-4zm-8 2a2 2 0 11-4 0 2 2 0 014 0z" />
                </svg>
              </button>
              {carrito.length > 0 && (
                <span className="absolute -top-2 -right-2 bg-pink-500 text-white text-xs font-bold rounded-full w-5 h-5 flex items-center justify-center">
                  {carrito.reduce((total, item) => total + item.cantidad, 0)}
                </span>
              )}
            </div>
            {usuario ? (
              <button
                onClick={handleCerrarSesion}
                className="text-white hover:text-pink-400 transition-colors text-sm"
              >
                {usuario.nombre}
              </button>
            ) : (
              <button
                onClick={() => setMostrarModalLogin(true)}
                className="text-white hover:text-pink-400 transition-colors text-sm"
              >
                Iniciar SesiÃ³n
              </button>
            )}
          </div>
        </div>
      </header>

      {/* Hero / Inicio Section inspirado en el boceto */}
      <section id="inicio" className="relative bg-linear-to-r from-purple-950 via-gray-900 to-gray-900 py-16">
        <div className="container mx-auto px-4">
          <div className="grid gap-10 md:grid-cols-[auto,1.2fr,auto] items-stretch">
            {/* Columna izquierda: acciones principales */}
            <div className="flex flex-col justify-between gap-8">
              <div className="space-y-3">
                <button
                  onClick={() =>
                    document.getElementById('destacados')?.scrollIntoView({ behavior: 'smooth' })
                  }
                  className="w-full md:w-40 bg-pink-500 hover:bg-pink-600 text-white font-semibold py-3 px-6 rounded-lg shadow-lg shadow-pink-500/30 transition-all duration-200"
                >
                  Ver catÃ¡logo
                </button>
                <button
                  onClick={() =>
                    document.getElementById('crear-prenda')?.scrollIntoView({ behavior: 'smooth' })
                  }
                  className="w-full md:w-40 bg-transparent border border-pink-400 text-pink-300 hover:bg-pink-500/10 font-semibold py-3 px-6 rounded-lg transition-all duration-200"
                >
                  Crear mi prenda
                </button>
              </div>
              <a
                href="#quienes-somos"
                className="text-sm tracking-[0.25em] uppercase text-gray-400 hover:text-pink-300 transition-colors"
              >
                Â¿QuiÃ©nes somos?
              </a>
            </div>

            {/* Columna central: destacados tipo boceto */}
            <div className="bg-gray-900/60 border border-gray-700 rounded-2xl p-6 md:p-8 flex flex-col gap-5">
              <h2 className="text-3xl md:text-4xl font-bold tracking-wide text-gray-100 mb-1">
                Destacados
              </h2>
              <div className="flex-1 flex flex-col gap-4">
                <div className="flex-1 border border-pink-500/70 rounded-xl p-4 flex items-center justify-between bg-gray-900/80">
                  <div>
                    <p className="text-sm uppercase tracking-[0.25em] text-gray-400 mb-1">
                      Talles
                    </p>
                    <p className="text-xl md:text-2xl font-semibold text-white">
                      Hasta 6XL
                    </p>
                  </div>
                  <div className="w-20 h-20 md:w-24 md:h-24 border-2 border-dashed border-pink-400/70 rounded-lg flex items-center justify-center text-4xl text-pink-300">
                    ðŸ‘•
                  </div>
                </div>

                <div className="flex-1 border border-purple-500/60 rounded-xl p-4 flex items-center justify-between bg-gray-900/80">
                  <div>
                    <p className="text-sm uppercase tracking-[0.25em] text-gray-400 mb-1">
                      Accesorios
                    </p>
                    <p className="text-xl md:text-2xl font-semibold text-white">
                      Sombreros Ãºnicos
                    </p>
                  </div>
                  <div className="w-20 h-20 md:w-24 md:h-24 border-2 border-dashed border-purple-400/70 rounded-lg flex items-center justify-center text-3xl text-purple-300">
                    ðŸŽ©
                  </div>
                </div>

                <div className="flex-1 border border-pink-300/70 rounded-xl p-4 flex items-center justify-between bg-gray-900/80">
                  <div>
                    <p className="text-sm uppercase tracking-[0.25em] text-gray-400 mb-1">
                      Telas
                    </p>
                    <p className="text-xl md:text-2xl font-semibold text-white">
                      Curadas para vos
                    </p>
                  </div>
                  <div className="w-20 h-20 md:w-24 md:h-24 border-2 border-dashed border-pink-200/80 rounded-lg flex items-center justify-center text-3xl text-pink-200">
                    ðŸ§µ
                  </div>
                </div>
              </div>
            </div>

            {/* Columna derecha: branding MERA'S */}
            <div className="flex flex-col items-end justify-between">
              <div className="text-right">
                <p className="text-xs uppercase tracking-[0.35em] text-gray-400 mb-2">
                  Inicio
                </p>
                <h2 className="text-5xl md:text-6xl font-extrabold leading-none bg-linear-to-b from-gray-50 to-gray-400 bg-clip-text text-transparent">
                  MERA&apos;S
                </h2>
                <p className="mt-3 text-sm md:text-base text-gray-300 tracking-[0.25em] uppercase">
                  wear your world
                </p>
              </div>
            </div>
          </div>
        </div>
      </section>

      {/* CategorÃ­as Section */}
      <section id="categorias" className="py-12 bg-gray-900">
        <div className="container mx-auto px-4">
          <div className="flex flex-col md:flex-row md:items-center md:justify-between mb-8">
            <h2 className="text-4xl font-bold mb-4 md:mb-0">CategorÃ­as</h2>
            <div className="flex gap-3 flex-wrap">
              <button
                onClick={() => setCategoriaFiltro('')}
                className={`px-4 py-2 rounded-lg font-semibold transition-colors ${
                  categoriaFiltro === '' 
                    ? 'bg-pink-500 text-white' 
                    : 'bg-gray-800 text-white hover:bg-gray-700'
                }`}
              >
                Todas
              </button>
              {categorias.map(categoria => (
                <button
                  key={categoria}
                  onClick={() => setCategoriaFiltro(categoria)}
                  className={`px-4 py-2 rounded-lg font-semibold transition-colors ${
                    categoriaFiltro === categoria 
                      ? 'bg-pink-500 text-white' 
                      : 'bg-gray-800 text-white hover:bg-gray-700'
                  }`}
                >
                  {categoria}
                </button>
              ))}
            </div>
          </div>
        </div>
      </section>

      {/* Destacados Section - catÃ¡logo */}
      <section id="destacados" className="py-12 bg-gray-900 border-t border-gray-800">
        <div className="container mx-auto px-4">
          <div className="flex flex-col md:flex-row md:items-center md:justify-between mb-8">
            <h2 className="text-4xl font-bold mb-4 md:mb-0">Destacados</h2>
            <div className="w-full md:w-auto">
              <input
                type="text"
                placeholder="Â¿QuÃ© estÃ¡s buscando?"
                value={busqueda}
                onChange={(e) => setBusqueda(e.target.value)}
                className="w-full md:w-80 px-4 py-2 bg-gray-800 border border-gray-700 rounded-lg text-white placeholder-gray-400 focus:outline-none focus:border-pink-500"
              />
            </div>
          </div>
          
          {prendasFiltradas.length === 0 ? (
            <div className="text-center py-12 text-gray-400">
              No se encontraron prendas
            </div>
          ) : (
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
              {prendasFiltradas.map((prenda) => (
                <PrendaCard
                  key={prenda.id}
                  prenda={prenda}
                  onEditar={handleEditar}
                  onComprar={handleComprar}
                />
              ))}
            </div>
          )}
        </div>
      </section>

      {/* Mensaje de notificaciÃ³n */}
      {mensaje.texto && (
        <div className={`fixed top-4 right-4 z-50 px-6 py-3 rounded-lg shadow-lg ${
          mensaje.tipo === 'success' ? 'bg-green-500' : 'bg-red-500'
        } text-white font-semibold`}>
          {mensaje.texto}
        </div>
      )}

      {/* Modales */}
      {mostrarModalEditar && prendaSeleccionada && (
        <ModalEditarPrenda
          prenda={prendaSeleccionada}
          onCerrar={() => {
            setMostrarModalEditar(false);
            setPrendaSeleccionada(null);
          }}
          onGuardar={handleGuardarEdicion}
        />
      )}

      {mostrarModalLogin && (
        <ModalLogin
          onCerrar={() => setMostrarModalLogin(false)}
          onLoginExitoso={handleLoginExitoso}
        />
      )}

      {mostrarAdminPanel && (
        <AdminPanel
          onCerrar={() => setMostrarAdminPanel(false)}
          onActualizar={cargarPrendas}
        />
      )}
    </div>
  );
}

export default App;
